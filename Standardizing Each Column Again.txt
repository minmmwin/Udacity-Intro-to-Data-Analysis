SOLUTION:

def standardize(df):
	return df.sub(df.mean(), axis='columns').div(df.std(ddof=0))

def standardize_rows(df):
	return df.sub(df.mean(axis=1), axis='index').div(df.std(ddof=0, axis=1), axis='index')
